\begin{codebox}
\Procname{$\proc{Init-LCS}(X,Y)$}
\li $m \gets \attrib{X}{length}$
\li $n \gets \attrib{Y}{length}$
\li \If $(m = 0)$ \Or $(n = 0)$
\li \Then
		\Return $0$
	\End
\li \For $i \gets 0$ \To $m$
\li \Do
		$L[i,0] \gets 0$
	\End
\li \For $j \gets 0$ \To $n$
\li \Do
		$L[0,j] \gets 0$
	\End
\li \For $i \gets 1$ \To $m$
\li \Do
		\For $i \gets 1$ \To $n$
\li 	\Do
			$L[i,j] \gets -1$
		\End
	\End
\li \Return $\proc{R-LCS}(X,Y,m,n)$
\end{codebox}
