\begin{codebox}
\Procname{$\proc{Rec-Sel}(S,f,i)$}
\li	$n \gets \attrib{S}{length}$
\li	$m \gets i + 1$
\li	\While $(m \leq n)$ \And $(s_n < f_i)$
\li	\Do
		$m \gets m + 1$
	\End
\li	\If $m \leq n$
\li	\Then
		\Return $\{a_m\} \cup \proc{Rec-Sel}(S,f,m)$
\li	\Else
\li		\Return $\emptyset$
	\End
\end{codebox}
