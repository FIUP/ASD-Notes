\begin{codebox}
\Procname{$\proc{Insert}(T,x)$}
\li $i \gets 0$
\li \Repeat
\li     $j \gets h(\attrib{x}{key},i)$
\li     \If $(T[j] = \const{nil})$ \Or $(T[j] = \const{deleted})$ 
        	\Comment posizione libera
\li     \Then
            $T[j] \gets x$
\li         \Return $j$
        \End
\li     $i \gets i + 1$
\li \Until $i = m$
\li \Error
\end{codebox}
