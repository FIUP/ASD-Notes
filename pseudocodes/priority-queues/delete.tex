\begin{codebox}
\Procname{$\proc{Delete}(A,i)$}
\li $\id{old} \gets A[i]$
\li $A[i] \gets A[\attrib{A}{heapsize}]$
\li $\attrib{A}{heapsize} \gets \attrib{A}{heapsize} - 1$
\li \If $\id{old} \leq A[i]$
\li \Then $\proc{Max-Heapify-Up}(A,i)$
\li \Else
\li		$\proc{Max-Heapify}(A,i)$
    \End
\end{codebox}
