\begin{codebox}
\Procname{$\proc{Merge}(A,p,q,r)$}
\li $n_1 \gets q-p+1$
    \Comment gli indici partono da 1
\li $n_2 \gets r-q$
\zi \Comment \kw{L} sotto-array sx, \kw{R} sotto-array dx
\li \For $i \gets 1$ \To $n_1$
\li \Do
        $L[i] \gets A[p+i-1]$
    \End
\li	\For $j \gets 1$ \To $n_2$
\li	\Do
		$R[j] \gets A[q+j]$
    \End
\li $L[n_1+1] \gets R[n_2+1] \gets \infty$
\li $i \gets j \gets 1$
\li \For $k \gets p$ \To $r$
\li \Do
    	\If $L[i] \leq R[j]$
\li     \Then
            $A[k] \gets L[i]$
\li         $i \gets i+1$
\li     \Else
        	\Comment \texttt{L[i]} $>$ \texttt{R[j]}
\li         $A[k] \gets R[j]$
\li         $j \gets j+1$
		\End
	\End
\end{codebox}
